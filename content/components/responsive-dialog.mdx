```tsx
"use client";

import { useState } from "react";
import { Button } from "@/components/ui/button";
import { ResponsiveDialog } from "@/registry/components/responsive-dialog";

export function ResponsiveDialogExample() {
  const [open, setOpen] = useState(false);

  return (
    <>
      <Button onClick={() => setOpen(true)}>Open Dialog</Button>
      <ResponsiveDialog
        isOpen={open}
        setIsOpen={setOpen}
        title="Edit profile"
        description="Make changes to your profile here. Click save when you're done."
      >
        <p className="text-sm text-muted-foreground">
          Dialog content goes here.
        </p>
      </ResponsiveDialog>
    </>
  );
}
````

## Installation

**pnpm**

```bash
pnpm dlx shadcn@latest add @chaesunbak/responsive-dialog
````

**npm**

```bash
npx shadcn@latest add @chaesunbak/responsive-dialog
```

## Usage

```tsx
import { ResponsiveDialog } from "@/components/responsive-dialog";
```

```tsx
<ResponsiveDialog
  isOpen={open}
  setIsOpen={setOpen}
  title="Edit profile"
  description="Make changes to your profile here. Click save when you're done."
>
  <button>open dialog</button>
</ResponsiveDialog>
```

## API Reference

### ResponsiveDialog

| Prop          | Type                                | Default     |
| ------------- | ----------------------------------- | ----------- |
| `isOpen`      | `boolean`                           | -           |
| `setIsOpen`   | `Dispatch<SetStateAction<boolean>>` | -           |
| `title`       | `string`                            | -           |
| `description` | `string`                            | `undefined` |
| `children`    | `ReactNode`                         | -           |
